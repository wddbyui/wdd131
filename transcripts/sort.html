Let's go over using a simple text sort.

                
    const simpleList = ["oranges", "grapes", "lemons", "apples", "Bananas", "watermelons", "coconuts", "broccoli", "mango"];

    let simpleSort = simpleList.sort();
    console.log(simpleSort);

Notice how 'Banana' is sorted first. This is because Unicode sorts capital letters before lowercase. Let's use .toLowerCase() to solve this issue. We are using .map to make a new array called lowerList. The .map function takes each array item and converts it to lowercase and the new array lowerList now contains all the same fruit but in lowercase. Then that list is sorted with .sort().

                
let lowerList = simpleList.map(function(fruit){
    return fruit.toLowerCase();
})

let lowerSort = lowerList.sort();
console.log(lowerSort);

Let's try sorting a number array. Notice how they are sorted with the unicode placing 10 and 12 before 9 and 8.

                
nums = [10, 12, 9, 8, 1];

let numSort = nums.sort();
console.log(numSort);

We will need to use a compare function to sort our numbers. It compares two of the array items (a and b) to each other again and again until they are all in the right order; either moving them up one (1) or down one (-1) or leaving them alone if they are equal (0). 

                
function compareFn(a,b) {
  if (a < b) {
    return -1;
  } else if (a > b) {
    return 1;
  }
 // a must be equal to b
 return 0;
}
               
let sortNums = nums.sort(compareFn);
console.log(sortNums);

We will also need the compare function when sorting properties of objects in an object array. For example if I wanted to sort these products objects alphabetically by the product name I would need to specify which property to use in the compare function.

                
const products = [
  {
    productName: "Wireless Mouse",
    price: 29.99
  },
  {
    productName: "Bluetooth Keyboard",
    price: 49.99
  },
  {
    productName: "Laptop Stand",
    price: 39.99
  }
];

function compareObjectFn(a,b) {
  if (a.productName < b.productName) {
    return -1;
  } else if (a.productName > b.productName) {
    return 1;
  }
 return 0;
}

let productSort = products.sort(compareObjectFn);
console.log(productSort);

If I wanted to sort by the price of the product. I'd change the property in the compare function. Then it sorts the objects in the array numerically by the price.

                
function compareObjectFn(a,b) {
  if (a.price < b.price) {
    return -1;
  } else if (a.price > b.price) {
    return 1;
  }
 // a must be equal to b
 return 0;
}

let productSort = products.sort(compareObjectFn);
console.log(productSort);
